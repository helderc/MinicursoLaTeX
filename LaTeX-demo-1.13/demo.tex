    \end{description}
  \end{abstract}

\begin{abstract}
  Este documento de \pageref{LastPage}~páginas é um exemplo de uso de

  usados: \usedpackages{}.

  \begin{description}
  \item[Palavras-chave] \docpalavraschave.
  \end{description}
\end{abstract}
